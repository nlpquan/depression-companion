## 🚀 Quick Start

```bash
# Clone the repository
git clone https://github.com/yourusername/depression-companion.git
cd depression-companion

# Create virtual environment
python -m venv venv
source venv/bin/activate  # Windows: venv\Scripts\activate

# Install dependencies
pip install -e ".[dev]"

# Run tests
pytest tests/ -v

# Run the pipeline
python -c "
from depression_companion.config import load_config
from depression_companion.pipeline.orchestrator import DepressionDetectionPipeline

config = load_config()
pipeline = DepressionDetectionPipeline(config)
result = pipeline.run(
    input_id='example',
    text='I have been feeling really down lately.'
)
print(f'Depression score: {result.predictions[\"depression_score\"]:.3f}')
"

# TESTING

# Run all tests
pytest tests/ -v

# Run with coverage
pytest tests/ -v --cov=src --cov-report=html

# Run specific test file
pytest tests/unit/test_audio_processor.py -v
